## 1. 对话功能

### 1.1 功能描述

用户在输入框输入消息，发送给 AI，AI 返回响应。整个流程包括：输入验证、上下文构建、消息发送、流式接收、响应渲染。

### 1.4 后端实现

```kotlin
/**
 * 聊天服务实现
 */
@Service(Service.Level.PROJECT)
class ChatServiceImpl(
    private val project: Project,
    private val sessionService: SessionService,
    private val contextService: ContextService,
    private val daemonBridge: DaemonBridgeService
) : ChatService {

    private val streamListeners = mutableSetOf<StreamListener>()
    private val currentRequests = mutableMapOf<String, MessageContext>()

    override fun sendMessage(request: SendMessageRequest): CompletableFuture<SendMessageResponse> {
        val future = CompletableFuture<SendMessageResponse>()

        // 1. 验证输入
        val validation = validateRequest(request)
        if (!validation.isValid) {
            future.complete(SendMessageResponse(
                success = false,
                messageId = "",
                error = validation.error
            ))
            return future
        }

        // 2. 创建消息ID
        val messageId = UUID.randomUUID().toString()
        val userMessage = Message(
            id = messageId,
            role = Role.USER,
            content = request.content,
            timestamp = Instant.now(),
            fileReferences = request.fileReferences,
            attachments = request.attachments
        )

        // 3. 添加用户消息到会话
        sessionService.addMessage(request.sessionId, userMessage)

        // 4. 构建上下文
        val context = contextService.buildContext(request.sessionId, userMessage)

        // 5. 创建AI消息占位
        val aiMessageId = UUID.randomUUID().toString()
        val aiMessage = Message(
            id = aiMessageId,
            role = Role.ASSISTANT,
            content = "",
            timestamp = Instant.now()
        )
        sessionService.addMessage(request.sessionId, aiMessage)

        // 6. 创建消息上下文
        val messageContext = MessageContext(
            messageId = aiMessageId,
            sessionId = request.sessionId,
            request = request,
            context = context,
            future = future
        )
        currentRequests[aiMessageId] = messageContext

        // 7. 发送到Daemon
        val callback = object : MessageCallback {
            override fun onChunk(chunk: String) {
                notifyListeners { it.onChunk(StreamChunk(aiMessageId, chunk, 0)) }
            }

            override fun onThinking(content: String) {
                notifyListeners { it.onThinking(ThinkingContent(aiMessageId, content, false)) }
            }

            override fun onToolUse(name: String, input: JsonObject) {
                val toolId = UUID.randomUUID().toString()
                notifyListeners {
                    it.onToolUse(ToolUseEvent(aiMessageId, toolId, name, input, ToolUseStatus.RUNNING))
                }
            }

            override fun onComplete(result: QueryResult) {
                val completion = MessageCompletion(
                    messageId = aiMessageId,
                    usage = result.usage,
                    stopReason = result.stopReason,
                    toolCalls = result.toolCalls.map { tc ->
                        ToolCallResult(tc.id, tc.name, tc.input, tc.output, ToolUseStatus.SUCCESS)
                    }
                )
                notifyListeners { it.onComplete(completion) }

                // 更新消息
                val updatedMessage = aiMessage.copy(
                    content = result.content,
                    usage = result.usage,
                    toolCalls = result.toolCalls
                )
                sessionService.updateMessage(request.sessionId, aiMessageId, updatedMessage)

                future.complete(SendMessageResponse(true, aiMessageId))
                currentRequests.remove(aiMessageId)
            }

            override fun onError(error: Throwable) {
                notifyListeners {
                    it.onError(StreamError(aiMessageId, "STREAM_ERROR", error.message ?: "Unknown error", true))
                }
                future.complete(SendMessageResponse(false, aiMessageId, error.message))
                currentRequests.remove(aiMessageId)
            }
        }

        daemonBridge.sendMessage(QueryRequest(
            sessionId = request.sessionId,
            messages = context.messages,
            options = QueryOptions(
                model = request.options.model,
                mode = request.options.mode.name.lowercase(),
                permissionMode = request.options.permissionMode.name.lowercase(),
                agent = request.options.agentId ?: "default",
                maxTokens = request.options.maxTokens,
                temperature = request.options.temperature
            )
        ), callback)

        return future
    }

    override fun registerStreamListener(listener: StreamListener) {
        streamListeners.add(listener)
    }

    override fun cancelCurrentMessage() {
        currentRequests.values.firstOrNull()?.let { context ->
            daemonBridge.interrupt()
        }
    }

    private fun notifyListeners(action: (StreamListener) -> Unit) {
        streamListeners.forEach { action(it) }
    }

    private fun validateRequest(request: SendMessageRequest): ValidationResult {
        if (request.content.isBlank() && request.attachments.isEmpty()) {
            return ValidationResult(false, "Message content is empty")
        }
        if (request.content.length > 100000) {
            return ValidationResult(false, "Message too long")
        }
        return ValidationResult(true)
    }

    data class ValidationResult(
        val isValid: Boolean,
        val error: String? = null
    )

    data class MessageContext(
        val messageId: String,
        val sessionId: String,
        val request: SendMessageRequest,
        val context: ConversationContext,
        val future: CompletableFuture<SendMessageResponse>
    )
}
```

### 1.5 前端实现

```kotlin
/**
 * 聊天面板实现
 */
class ChatPanel(
    private val project: Project
) : JPanel(BorderLayout()), StreamListener {

    private val chatService = project.getService(ChatService::class.java)
    private val messageArea = MessageAreaPanel()
    private val inputArea = InputAreaPanel()
    private val statusBar = StatusBarPanel()

    private var currentMessageId: String? = null
    private var accumulatedContent = StringBuilder()

    init {
        // 注册流式监听器
        chatService.registerStreamListener(this)

        // 构建UI
        add(messageArea, BorderLayout.CENTER)
        add(inputArea, BorderLayout.SOUTH)
        add(statusBar, BorderLayout.NORTH)

        // 绑定发送事件
        inputArea.onSend = ::handleSend
    }

    private fun handleSend() {
        val content = inputArea.getContent()
        val fileRefs = inputArea.getFileReferences()
        val attachments = inputArea.getAttachments()

        if (content.isBlank() && attachments.isEmpty()) {
            showError("Message cannot be empty")
            return
        }

        // 显示用户消息
        val userMessage = UserMessageCard(
            content = content,
            fileReferences = fileRefs,
            attachments = attachments
        )
        messageArea.addMessage(userMessage)

        // 清空输入框
        inputArea.clear()

        // 发送消息
        val request = SendMessageRequest(
            sessionId = getCurrentSessionId(),
            content = content,
            fileReferences = fileRefs,
            attachments = attachments,
            options = getCurrentOptions()
        )

        // 显示加载状态
        currentMessageId = null
        accumulatedContent.clear()
        statusBar.setState(Status.THINKING)

        chatService.sendMessage(request)
            .whenComplete { response, throwable ->
                if (throwable != null) {
                    showError("Failed to send message: ${throwable.message}")
                    statusBar.setState(Status.ERROR)
                } else if (!response.success) {
                    showError(response.error ?: "Unknown error")
                    statusBar.setState(Status.ERROR)
                }
            }
    }

    override fun onChunk(chunk: StreamChunk) {
        // 在EDT线程更新UI
        ApplicationManager.getApplication().invokeLater {
            currentMessageId = chunk.messageId
            accumulatedContent.append(chunk.content)

            // 更新或创建AI消息卡片
            messageArea.updateMessage(chunk.messageId, accumulatedContent.toString())
            statusBar.setState(Status.STREAMING)
        }
    }

    override fun onThinking(thinking: ThinkingContent) {
        ApplicationManager.getApplication().invokeLater {
            messageArea.showThinking(thinking.messageId, thinking.content)
        }
    }

    override fun onToolUse(toolUse: ToolUseEvent) {
        ApplicationManager.getApplication().invokeLater {
            messageArea.showToolUse(toolUse)
        }
    }

    override fun onComplete(completion: MessageCompletion) {
        ApplicationManager.getApplication().invokeLater {
            messageArea.finalizeMessage(completion.messageId)
            statusBar.updateUsage(completion.usage)
            statusBar.setState(Status.IDLE)
        }
    }

    override fun onError(error: StreamError) {
        ApplicationManager.getApplication().invokeLater {
            if (error.isFatal) {
                messageArea.showError(error.messageId, error.message)
            } else {
                messageArea.showWarning(error.messageId, error.message)
            }
            statusBar.setState(Status.ERROR)
        }
    }
}
```

```kotlin
/**
 * 流式输出管理器实现
 */
@Service(Service.Level.PROJECT)
class StreamManagerImpl : StreamManager {

    private val streams = mutableMapOf<String, StreamContext>()
    private val consumers = mutableMapOf<String, (String) -> Unit>()

    override fun startStream(messageId: String, consumer: (String) -> Unit) {
        streams[messageId] = StreamContext(
            messageId = messageId,
            content = StringBuilder(),
            startTime = Instant.now()
        )
        consumers[messageId] = consumer
    }

    override fun appendText(messageId: String, text: String) {
        val context = streams[messageId] ?: return
        context.content.append(text)

        // 通知消费者
        consumers[messageId]?.invoke(text)
    }

    override fun completeStream(messageId: String) {
        val context = streams[messageId] ?: return
        context.endTime = Instant.now()
        context.isComplete = true
    }

    override fun cancelStream(messageId: String) {
        streams.remove(messageId)
        consumers.remove(messageId)
    }

    override fun getStreamContent(messageId: String): String {
        return streams[messageId]?.content?.toString() ?: ""
    }

    data class StreamContext(
        val messageId: String,
        val content: StringBuilder,
        val startTime: Instant,
        var endTime: Instant? = null,
        var isComplete: Boolean = false
    )
}
```

```kotlin
/**
 * 流式消息卡片
 */
class StreamingMessageCard(
    private val messageId: String
) : MessageCard() {

    private val textPane = JTextPane().apply {
        isEditable = false
        contentType = "text/html"
    }

    private val contentBuffer = StringBuilder()
    private var lastUpdateTime = 0L
    private val updateInterval = 16L // ~60fps

    init {
        add(textPane, BorderLayout.CENTER)
    }

    /**
     * 追加文本 (节流更新)
     */
    fun appendText(text: String) {
        contentBuffer.append(text)

        val now = System.currentTimeMillis()
        if (now - lastUpdateTime >= updateInterval) {
            updateDisplay()
            lastUpdateTime = now
        }
    }

    /**
     * 强制更新显示
     */
    fun updateDisplay() {
        val html = MarkdownRenderer.render(contentBuffer.toString())
        textPane.text = html

        // 滚动到底部
        SwingUtilities.invokeLater {
            textPane.caretPosition = textPane.document.length
        }
    }

    /**
     * 完成流式
     */
    fun complete() {
        updateDisplay()
        // 添加完整消息的操作按钮
        addActionButtonPanel()
    }
}
```

```kotlin
/**
 * Provider服务实现

@Service(Service.Level.APPLICATION)
class ProviderServiceImpl(
    private val configService: ConfigService,
    private val daemonBridge: DaemonBridgeService,
    private val application: Application
) : ProviderService {

    private val listeners = mutableListOf<ProviderChangeListener>()
    private var activeProviderId: String = DEFAULT_PROVIDER_ID

    companion object {
        const val DEFAULT_PROVIDER_ID = "claude"
    }

    init {
        // 加载活跃Provider
        loadActiveProvider()
    }

    override fun getAllProviders(): List<ProviderConfig> {
        val config = configService.getState()
        return config.providers.toList()
    }

    override fun getActiveProvider(): ProviderConfig {
        val providers = getAllProviders()
        return providers.firstOrNull { it.id == activeProviderId }
            ?: providers.firstOrNull { it.id == DEFAULT_PROVIDER_ID }
            ?: createDefaultClaudeProvider()
    }

    override fun addProvider(config: ProviderConfig): Boolean {
        val state = configService.getState()

        // 检查ID是否重复
        if (state.providers.any { it.id == config.id }) {
            return false
        }

        state.providers.add(config)
        configService.saveSettings()

        // 通知监听器
        notifyListeners { it.onProviderAdded(config) }

        return true
    }

    override fun updateProvider(config: ProviderConfig): Boolean {
        val state = configService.getState()
        val index = state.providers.indexOfFirst { it.id == config.id }

        if (index < 0) {
            return false
        }

        // 保留isActive状态
        val wasActive = state.providers[index].isActive
        state.providers[index] = config.copy(isActive = wasActive)
        configService.saveSettings()

        // 通知监听器
        notifyListeners { it.onProviderUpdated(config) }

        // 如果是活跃Provider，通知Daemon
        if (config.id == activeProviderId) {
            notifyDaemonProviderChange(config)
        }

        return true
    }

    override fun removeProvider(providerId: String): Boolean {
        val state = configService.getState()

        // 不能删除内置Provider
        val provider = state.providers.firstOrNull { it.id == providerId }
            ?: return false

        if (provider.isBuiltIn) {
            return false
        }

        // 如果删除的是活跃Provider，切换到默认
        val wasActive = provider.isActive
        state.providers.removeIf { it.id == providerId }

        if (wasActive) {
            switchProvider(DEFAULT_PROVIDER_ID)
        }

        configService.saveSettings()

        // 通知监听器
        notifyListeners { it.onProviderRemoved(providerId) }

        return true
    }

    override fun switchProvider(providerId: String): Boolean {
        val providers = getAllProviders()
        val newProvider = providers.firstOrNull { it.id == providerId }
            ?: return false

        val oldProvider = getActiveProvider()

        // 更新活跃状态
        val state = configService.getState()
        state.providers.forEach { it.isActive = (it.id == providerId) }
        activeProviderId = providerId
        configService.saveSettings()

        // 通知Daemon
        notifyDaemonProviderChange(newProvider)

        // 通知监听器
        notifyListeners { it.onProviderChanged(oldProvider, newProvider) }

        return true
    }

    override suspend fun validateProvider(config: ProviderConfig): ValidationResult {
        return withContext(Dispatchers.IO) {
            try {
                // 根据类型验证
                when (config.type) {
                    ProviderType.CLAUDE -> validateClaudeProvider(config)
                    ProviderType.OPENROUTER -> validateOpenRouterProvider(config)
                    ProviderType.CUSTOM -> validateCustomProvider(config)
                }
            } catch (e: Exception) {
                ValidationResult(false, e.message)
            }
        }
    }

    override fun getAvailableModels(providerId: String): List<ModelInfo> {
        return when (providerId) {
            "claude" -> ClaudeModels.ALL
            "openrouter" -> OpenRouterModels.getAvailableModels()
            else -> getCustomModels(providerId)
        }
    }

    override fun addProviderChangeListener(listener: ProviderChangeListener) {
        application.invokeLater {
            listeners.add(listener)
        }
    }

    override fun removeProviderChangeListener(listener: ProviderChangeListener) {
        application.invokeLater {
            listeners.remove(listener)
        }
    }

    private suspend fun validateClaudeProvider(config: ProviderConfig): ValidationResult {
        // 调用Claude API验证
        val client = OkHttpClient()
        val request = Request.Builder()
            .url(config.endpoint)
            .addHeader("x-api-key", config.apiKey)
            .addHeader("anthropic-version", "2023-06-01")
            .post(
                "{\"model\":\"${config.defaultModel}\",\"max_tokens\":1,\"messages\":[{\"role\":\"user\",\"content\":\"hi\"}]}"
                .toRequestBody("application/json".toMediaType())
            )
            .build()

        val response = client.newCall(request).execute()

        return if (response.isSuccessful) {
            ValidationResult(true, null, getAvailableModels(config.id).firstOrNull())
        } else {
            ValidationResult(false, "API Key validation failed: ${response.code}")
        }
    }

    private fun notifyDaemonProviderChange(provider: ProviderConfig) {
        daemonBridge.sendCommand(mapOf(
            "type" to "switch_provider",
            "provider" to provider.id,
            "config" to mapOf(
                "apiKey" to provider.apiKey,
                "endpoint" to provider.endpoint
            )
        ))
    }

    private fun notifyListeners(action: (ProviderChangeListener) -> Unit) {
        application.invokeLater {
            listeners.toList().forEach { action(it) }
        }
    }

    private fun loadActiveProvider() {
        val state = configService.getState()
        val active = state.providers.firstOrNull { it.isActive }
        if (active != null) {
            activeProviderId = active.id
        }
    }

    private fun createDefaultClaudeProvider(): ProviderConfig {
        return ProviderConfig(
            id = "claude",
            name = "Claude (Anthropic)",
            type = ProviderType.CLAUDE,
            apiKey = "",
            endpoint = "https://api.anthropic.com",
            defaultModel = "claude-sonnet-4-20250514",
            isActive = true,
            isBuiltIn = true
        )
    }
}
```

```kotlin
/**
 * 提示词增强服务实现
 */
@Service(Service.Level.PROJECT)
class PromptEnhancementServiceImpl(
    private val project: Project
) : PromptEnhancementService {

    private val rules = mutableListOf<EnhancementRule>()
    private val history = mutableListOf<EnhancementRecord>()

    init {
        // 注册默认规则
        registerDefaultRules()
    }

    override suspend fun enhancePrompt(
        originalPrompt: String,
        context: EnhancementContext
    ): EnhancementResult = withContext(Dispatchers.IO) {

        val changes = mutableListOf<EnhancementChange>()
        var enhancedPrompt = originalPrompt
        val appliedRules = mutableListOf<String>()

        // 按优先级应用规则
        rules
            .filter { it.isEnabled }
            .sortedBy { it.priority }
            .forEach { rule ->
                try {
                    val change = rule.apply(enhancedPrompt, context)
                    if (change != null) {
                        changes.add(change)
                        enhancedPrompt = change.enhanced
                        appliedRules.add(rule.id)
                    }
                } catch (e: Exception) {
                    // 规则应用失败，跳过
                }
            }

        // 计算置信度
        val confidence = calculateConfidence(changes, context)

        EnhancementResult(
            enhancedPrompt = enhancedPrompt,
            changes = changes,
            confidence = confidence,
            appliedRules = appliedRules
        )
    }

    override fun getEnhancementHistory(): List<EnhancementRecord> {
        return history.toList()
    }

    override fun configureRules(rules: List<EnhancementRule>) {
        this.rules.clear()
        this.rules.addAll(rules)
    }

    private fun registerDefaultRules() {
        // 规则1: 添加文件上下文
        rules.add(EnhancementRule(
            id = "add_file_context",
            name = "添加文件上下文",
            description = "自动将引用的文件内容添加到提示词",
            isEnabled = true,
            priority = 1
        ) { prompt, context ->
            if (context.fileReferences.isNotEmpty()) {
                val fileContext = buildFileContext(context.fileReferences)
                EnhancementChange(
                    type = ChangeType.CONTEXT_ADDED,
                    description = "添加了 ${context.fileReferences.size} 个文件的上下文",
                    original = prompt,
                    enhanced = "$prompt\n\n$fileContext"
                )
            } else null
        })

        // 规则2: 添加选中文本
        rules.add(EnhancementRule(
            id = "add_selected_text",
            name = "添加选中文本",
            description = "自动将编辑器中选中的文本添加到提示词",
            isEnabled = true,
            priority = 2
        ) { prompt, context ->
            if (!context.selectedText.isNullOrBlank()) {
                EnhancementChange(
                    type = ChangeType.CONTEXT_ADDED,
                    description = "添加了选中的代码",
                    original = prompt,
                    enhanced = "$prompt\n\n选中的代码:\n```\n${context.selectedText}\n```"
                )
            } else null
        })

        // 规则3: 代码优化格式化
        rules.add(EnhancementRule(
            id = "code_optimization_format",
            name = "代码优化格式",
            description = "检测代码优化请求，添加标准格式",
            isEnabled = true,
            priority = 3
        ) { prompt, context ->
            val keywords = listOf("优化", "改进", "重构", "refactor", "optimize")
            if (keywords.any { prompt.contains(it, ignoreCase = true) }) {
                EnhancementChange(
                    type = ChangeType.FORMAT_FIXED,
                    description = "添加了代码优化标准格式",
                    original = prompt,
                    enhanced = """
                        |请帮我优化以下代码：
                        |
                        |要求：
                        |1. 保持原有功能
                        |2. 提高可读性
                        |3. 改进性能
                        |4. 遵循最佳实践
                        |
                        |原始请求: $prompt
                        |
                        |请提供:
                        |- 优化后的代码
                        |- 修改说明
                        |- 性能对比
                    """.trimMargin()
                )
            } else null
        })

        // 规则4: 添加语言标识
        rules.add(EnhancementRule(
            id = "add_language_hint",
            name = "添加语言标识",
            description = "根据文件类型添加编程语言标识",
            isEnabled = true,
            priority = 4
        ) { prompt, context ->
            context.currentFile?.let { file ->
                val language = file.fileType.name
                EnhancementChange(
                    type = ChangeType.KEYWORDS_ADDED,
                    description = "添加了编程语言标识",
                    original = prompt,
                    enhanced = "[$language] $prompt"
                )
            }
        })

        // 规则5: 结构化复杂请求
        rules.add(EnhancementRule(
            id = "structure_complex_request",
            name = "结构化复杂请求",
            description = "将复杂请求结构化",
            isEnabled = true,
            priority = 5
        ) { prompt, context ->
            if (prompt.length > 200 && !prompt.contains("\n")) {
                // 尝试识别句子
                val sentences = prompt.split(Regex("[。.!?！？]"))
                    .map { it.trim() }
                    .filter { it.isNotEmpty() }

                if (sentences.size > 1) {
                    val structured = sentences.joinToString("\n") { "• $it" }
                    EnhancementChange(
                        type = ChangeType.STRUCTURED,
                        description = "将请求结构化",
                        original = prompt,
                        enhanced = "我的需求:\n\n${structured}\n\n请逐一处理。"
                    )
                } else null
            } else null
        })
    }

    private fun buildFileContext(references: List<FileReference>): String {
        return buildString {
            append("引用的文件:\n")
            references.forEach { ref ->
                append("• ${ref.path}")
                if (ref.startLine != null) {
                    append(" (行 ${ref.startLine}-${ref.endLine ?: ref.startLine})")
                }
                append("\n")
            }
        }
    }

    private fun calculateConfidence(
        changes: List<EnhancementChange>,
        context: EnhancementContext
    ): Double {
        // 基于变更类型和上下文计算置信度
        var confidence = 0.5

        changes.forEach { change ->
            confidence += when (change.type) {
                ChangeType.CONTEXT_ADDED -> 0.15
                ChangeType.FORMAT_FIXED -> 0.10
                ChangeType.STRUCTURED -> 0.10
                ChangeType.KEYWORDS_ADDED -> 0.05
                ChangeType.CLARITY_IMPROVED -> 0.10
            }
        }

        // 有文件引用时提高置信度
        if (context.fileReferences.isNotEmpty()) {
            confidence += 0.10
        }

        return confidence.coerceAtMost(1.0)
    }
}
```

### 8.5 后端实现

```kotlin
/**
 * 国际化服务实现
 */
@Service(Service.Level.APPLICATION)
class I18nServiceImpl(
    private val application: Application
) : I18nService {

    private val listeners = mutableListOf<LanguageChangeListener>()
    private var currentLanguage = Language.AUTO
    private val bundleCache = mutableMapOf<String, ResourceBundle>()

    companion object {
        const val BUNDLE_BASE = "messages.CCAssistantBundle"
        const val COMMANDS_BUNDLE_BASE = "messages.CommandsBundle"
    }

    override fun getMessage(key: String, vararg params: Any): String {
        val language = getEffectiveLanguage()
        val bundle = getBundle(language)

        return try {
            val pattern = bundle.getString(key)
            if (params.isEmpty()) {
                pattern
            } else {
                MessageFormat.format(pattern, *params)
            }
        } catch (e: MissingResourceException) {
            // 返回键本身
            key
        }
    }

    override fun getCurrentLanguage(): Language {
        return currentLanguage
    }

    override fun setLanguage(language: Language): Boolean {
        val oldLanguage = currentLanguage
        currentLanguage = language

        // 清除缓存
        bundleCache.clear()

        // 保存配置
        saveLanguageToConfig()

        // 通知监听器
        application.invokeLater {
            listeners.forEach { it.onLanguageChanged(oldLanguage, language) }
        }

        return true
    }

    override fun getAvailableLanguages(): List<Language> {
        return Language.values().toList()
    }

    override fun addLanguageChangeListener(listener: LanguageChangeListener) {
        application.invokeLater {
            listeners.add(listener)
        }
    }

    private fun getEffectiveLanguage(): Language {
        return if (currentLanguage == Language.AUTO) {
            detectLanguage()
        } else {
            currentLanguage
        }
    }

    private fun detectLanguage(): Language {
        val ideLocale = Locale.getDefault()
        return when {
            ideLocale.language == "zh" -> Language.ZH_CN
            ideLocale.language == "ja" -> Language.JA
            ideLocale.language == "ko" -> Language.KO
            else -> Language.EN
        }
    }

    private fun getBundle(language: Language): ResourceBundle {
        val bundleName = when (language) {
            Language.AUTO -> BUNDLE_BASE
            Language.ZH_CN -> "${BUNDLE_BASE}_zh"
            Language.EN -> BUNDLE_BASE
            Language.JA -> "${BUNDLE_BASE}_ja"
            Language.KO -> "${BUNDLE_BASE}_ko"
        }

        return bundleCache.getOrPut(bundleName) {
            ResourceBundle.getBundle(bundleName)
        }
    }

    private fun saveLanguageToConfig() {
        val config = application.getService(ConfigService::class.java)?.getState()
        config?.language = currentLanguage.code
    }
}

/**
 * 扩展函数
 */
fun Component.message(key: String, vararg params: Any): String {
    val service = service<I18nService>()
    return service.getMessage(key, *params)
}
```
